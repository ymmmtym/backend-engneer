    class User():

    def __init__(self, name="", age=0):
        self.name = name
        self.age = age

    def say_hello(self):
        print("Hello, my name is " + self.name)

    def __str__(self):
        return "name:" + self.name + " age:" + str(self.age)

    def __bool__(self):
        return True


user = User("hoge", 20)
str(user)
# 'name:hoge age:20'

bool(user)
# True
